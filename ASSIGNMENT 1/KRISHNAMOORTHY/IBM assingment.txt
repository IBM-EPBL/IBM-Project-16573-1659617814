float temp;
int tempPin = 0;

void setup() {
   Serial.begin(9600);
  pinMode(7,OUTPUT);
}

void loop() {
   temp = analogRead(tempPin);
   temp = temp * 0.488285;
   Serial.print("TEMPERATURE = ");
   Serial.print(temp); 
   Serial.print("*C");
   Serial.println();
  if(temp>50)
  {
    digitalWrite(7,HIGH);
  }
  else
  {
    digitalWrite(7,LOW);
  }
   delay(1000); 
}
